import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int m = sc.nextInt();
        int n = sc.nextInt();

        if (m < n) {
            System.out.println("a < b");
        } else if (m > n) {
            System.out.println("a > b");
        } else {
            System.out.println("a == b");
        }
    }
}

import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int T = sc.nextInt();

        for (int i = 0;i < T;i++) {
            int N = sc.nextInt();
            int X = sc.nextInt();

            int packets = 0;

            while (X < N) {
                X = X + 4;  
                packets++;  
            }

            System.out.println(packets);
        }
        sc.close();
    }
}

import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int T = sc.nextInt();

       for (int i=0;i<T;i++) {
            int N = sc.nextInt();
            int X = sc.nextInt();

            int slices = N * X;
            int pizzas = 0;

            while (slices > 0) {
                slices -= 4;
                pizzas++;
            }

            System.out.println(pizzas);
        }
        sc.close();
    }
}
